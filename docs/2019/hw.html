Problem #1

$$ "Assume " X ~ "Uniform"[a, b] " then " "Var"[X] = (b - a)^2 / 12 $$
$$ Var[X] $$
$$ = E[X^2] - E[X]^2 $$
$$ = [int_b^a j^2 Pr(X=j) \ dj] - ((a + b) / 2)^2 $$
$$ = [int_b^a j^2 (1 / (b - a)) dj] - ((a + b) / 2)^2 $$
$$ = (1 / (b - a)) \int_b^a j^2 dj - ((a + b) / 2)^2 $$
$$ = (1 / (b - a)) [1/3 j^3 + C]_a^b - ((a + b) / 2)^2 $$
$$ = (b - a)^2 / 12 $$
$$ $$

a

$$ Var[X] = (2k - 0)^2 / 12 = (4k^2) / 12 = k^2/3 $$

b

$$ Var[X] = (k - (-k))^2 / 12 = (4k^2) / 12 = k^2/3 $$

c

$$ Var[X + t] $$
$$ = E[((X + t) - E[X + t])^2] $$
$$ = E[(X + t - E[X] - E[t])^2] $$
$$ = E[(X + t - E[X] - t)^2] $$
$$ = E[(X - E[X])^2] = Var[X] $$

Problem #2

$$ "Independence" => E[X]E[Y] = E[XY] $$
$$ $$
$$ Var[X - Y] $$
$$ = E[(X - Y)^2] - E[X - Y]^2 $$
$$ = E[X^2 - 2XY + Y^2] - (E[X] - E[Y])^2 $$
$$ = E[X^2] - 2E[XY] + E[Y^2] - (E[X]^2 - 2E[X]E[Y] + E[Y]^2) $$
$$ = (E[X^2] - E[X]^2) + (E[Y^2] - E[Y]^2)$$
$$ = Var[X] + Var[Y] $$

Problem #3

$$ "premise. " Pr(X_i = 1) = 1//n $$
$$ "premise. " E[X] = n * (1//n) = 1 $$
$$ "premise. " E[X^2] $$
$$ = E[(sum_(i=1)^n X_i)^2] = E[(sum_(i=1)^n X_i)(sum_(j=1)^n X_j)] $$
$$ = E[sum_(i=1)^n sum_(j=1)^n X_i X_j] = sum_(i=1)^n sum_(j=1)^n E[X_i X_j] $$
$$ = (sum_(i=1)^n E[X_i]) + (sum_(i=1)^n sum_(j != i) E[X_i X_j]) $$
$$ = 1 + (sum_(i=1)^n sum_(j != i) Pr(X_i=1 nn X_j=1)) $$
$$ = 1 + (sum_(i=1)^n sum_(j != i) ((n-2)!) / (n!)) $$
$$ = 1 + (sum_(i=1)^n sum_(j != i) 1/(n (n-1))) $$
$$ = 1 + (1/(n (n-1)) sum_(i=1)^n sum_(j != i) 1) $$
$$ = 1 + 1 = 2 $$
$$ $$
$$ $$
$$ Var[X] $$
$$ = E[X^2] - E[X]^2 $$
$$ = 2 - 1 $$
$$ = 1 $$

Problem #4

a

$$ E[Z_i] $$
$$ = Pr(Z_i = 1) " where " Z_i " is indicator of edge (u, v)" $$
$$ = Pr("u is pressed, v is pressed") + Pr("u is not pressed, v is not pressed") $$
$$ = 1/4 + 1/4 = 1/2 $$

b

$$ E[Z_i Z_j] $$
$$ = sum_((i, j) in S) i j * Pr(Z_i=i, Z_j=j) " where S={(0, 0), (0, 1), (1, 0), (1, 1)}" $$
$$ = 1 * Pr(Z_i=1, Z_j=1) $$
$$ = 1 * (1/2 * 1/2)$$
$$ = 1/4 $$

c

$$ Cov[Z_i, Z_j] $$
$$ = E[Z_i Z_j] - E[Z_i]E[Z_j] $$
$$ = 1/4 - 1/2 * 1/2 = 0 $$

d

$$ "premise. " Var[Z_i] = E[Z_i^2] = E[Z_i]^2 - 1^2 * 1/2 - (1/2)^2 = 1/4 $$
$$ $$
$$ Pr(abs(Z - E[Z]) >= n) $$
$$ <= (Var[Z]) / n^2 $$
$$ = 1/n^2 ( sum_(i=1)^n Var[Z_i] + 2 sum_(i=1)^n sum_(j != i) Cov[Z_i, Z_j]) $$
$$ = 1/n^2 (1/4 - 0) $$
$$ = 1/(4n^2) $$

Problem #5

$$ lim_(n->oo) Pr(abs(X_1 + X_2 + ... + X_n) > n epsilon) $$
$$ <= lim_(n->oo) 1/(n^2 epsilon^2) Var[X_1 + X_2 + ... + X_n] $$
$$ = lim_(n->oo) 1/(n^2 epsilon^2) (n * sigma^2) $$
$$ = lim_(n->oo) (sigma^2) / (n epsilon^2) $$
$$ = 0 $$
